    üß† models.py

    from sqlmodel import SQLModel, Field, create_engine, Session, select
from typing import Optional
import json

DATABASE_URL = "postgresql://user:password@localhost:5432/tasksdb"

engine = create_engine(DATABASE_URL)

class TaskModel(SQLModel, table=True):
    id: Optional[int] = Field(default=None, primary_key=True)
    payload: str
    status: str = "pending"
    retry_count: int = 0
    celery_task_id: Optional[str] = None
    last_error: Optional[str] = None

def init_db():
    SQLModel.metadata.create_all(engine)

def get_session():
    return Session(engine)

    üöÄ main.py (FastAPI App)

    from fastapi import FastAPI, HTTPException
from models import TaskModel, get_session, init_db
from celery_worker import my_task
import json

app = FastAPI()

init_db()

@app.post("/process/")
def process_task(data: dict):
    session = get_session()
    task = TaskModel(payload=json.dumps(data))
    session.add(task)
    session.commit()

    try:
        result = my_task.apply_async(args=[data])
        task.status = "queued"
        task.celery_task_id = result.id
        session.add(task)
        session.commit()
    except Exception as e:
        task.last_error = str(e)
        session.commit()

    return {"task_id": task.id, "status": task.status}

    üêá celery_worker.py

    from celery import Celery
import time

# Broker URL (use Redis or RabbitMQ)
app = Celery("worker", broker="redis://localhost:6379/0")

@app.task
def my_task(data):
    print(f"Processing: {data}")
    time.sleep(2)
    return {"result": "done", "input": data}

    üîÅ retry_worker.py (Retry Failed Tasks)

    from models import TaskModel, get_session
from celery_worker import my_task
import json

MAX_RETRIES = 5

def retry_pending_tasks():
    session = get_session()
    tasks = session.query(TaskModel).filter(TaskModel.status == "pending").all()

    for task in tasks:
        if task.retry_count >= MAX_RETRIES:
            print(f"Giving up on task {task.id}")
            continue

        try:
            payload = json.loads(task.payload)
            result = my_task.apply_async(args=[payload])
            task.status = "queued"
            task.celery_task_id = result.id
        except Exception as e:
            task.retry_count += 1
            task.last_error = str(e)
        finally:
            session.add(task)
            session.commit()

if __name__ == "__main__":
    retry_pending_tasks()

    ‚úÖ To Run It All

    1. Start Redis: bash docker run -p 6379:6379 redis

    2. Start Celery Worker: bash celery -A celery_worker.app worker --loglevel=info

    3. Run FastAPI: bash uvicorn main:app --reload

    4. Manually run retry: bash python retry_worker.py

    üëâ You can set up a cron job or use APScheduler to run retry_worker.py periodically.
