    Installation Commands

    #!/bin/bash

# Install pyenv (Python Version Management)
echo "Installing pyenv..."
curl https://pyenv.run | bash
echo 'export PYENV_ROOT="$HOME/.pyenv"' >> ~/.bashrc
echo 'command -v pyenv >/dev/null || export PATH="$PYENV_ROOT/bin:$PATH"' >> ~/.bashrc
echo 'eval "$(pyenv init -)"' >> ~/.bashrc
source ~/.bashrc

# Install Python (latest stable version) using pyenv
echo "Installing Python via pyenv..."
pyenv install 3.11.5  # Replace with the desired version
pyenv global 3.11.5

# Install Just (Project-specific commands)
echo "Installing Just..."
curl --proto '=https' --tlsv1.2 -sSf https://just.systems/install.sh | bash -s -- --to ~/.local/bin
